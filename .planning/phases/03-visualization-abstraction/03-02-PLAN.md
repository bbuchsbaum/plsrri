```r
describe("brain_viewer_server", {

  it("calls renderer with filter values", {
    mock_renderer <- MockBrainRenderer$new()
    mock_result <- make_test_pls_result()

    testServer(brain_viewer_server, {
      session$flushReact()

      # Verify render was called
      expect_gte(length(mock_renderer$render_calls), 1)

      # Check default parameters
      last_call <- mock_renderer$render_calls[[length(mock_renderer$render_calls)]]
      expect_equal(last_call$lv, 1L)
      expect_equal(last_call$what, "bsr")
      expect_equal(last_call$threshold, 3.0)
    }, args = list(
      result_rv = reactive(mock_result),
      filters = list(
        lv = reactive(1L),
        bsr_threshold = reactive(3.0),
        what = reactive("bsr"),
        lag = reactive(NULL)
      ),
      renderer = mock_renderer
    ))
  })

  it("passes updated filter values to renderer", {
    mock_renderer <- MockBrainRenderer$new()
    mock_result <- make_test_pls_result()

    testServer(brain_viewer_server, {
      session$flushReact()
      mock_renderer$reset_calls()  # Clear initial render

      # Change LV via filter reactive (simulate filter_bar change)
      # Note: We need to update the filters reactive - this may require
      # using reactiveVal in test setup instead of reactive()
    }, args = list(
      result_rv = reactive(mock_result),
      filters = list(
        lv = reactive(2L),
        bsr_threshold = reactive(4.0),
        what = reactive("salience"),
        lag = reactive(NULL)
      ),
      renderer = mock_renderer
    ))

    # Verify updated values
    last_call <- mock_renderer$render_calls[[length(mock_renderer$render_calls)]]
    expect_equal(last_call$lv, 2L)
    expect_equal(last_call$what, "salience")
    expect_equal(last_call$threshold, 4.0)
  })

  it("handles NULL result gracefully", {
    mock_renderer <- MockBrainRenderer$new()

    testServer(brain_viewer_server, {
      session$flushReact()
      # With NULL result, renderer should NOT be called
      expect_equal(length(mock_renderer$render_calls), 0)
    }, args = list(
      result_rv = reactive(NULL),
      filters = list(
        lv = reactive(1L),
        bsr_threshold = reactive(3.0),
        what = reactive("bsr"),
        lag = reactive(NULL)
      ),
      renderer = mock_renderer
    ))
  })

  it("view mode toggle updates renderer call", {
```
